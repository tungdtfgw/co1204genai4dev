Stateful Chat Example

  • The chat object maintains the conversation history, enabling the model to answer follow-up questions in context.
stateful_chat.py Python 
from google import genai
from intro_gemini.utils import get_key

client = genai.Client(api_key=get_key())
# Create a chat session
chat = client.chats.create(model='gemini-2.0-flash')
while True:
    question = input('[User]: ')
    if question in ('exit', 'quit'):
        break
    response = chat.send_message(message=question)
    print('[GenAI]:', response.text)
# Inspect history
for message in chat.get_history():
    print(f'{message.role}: {message.parts[0].text}')
