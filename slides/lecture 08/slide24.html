Advanced Agent example

  • Load credentials: load_dotenv() pulls secrets from the local .env.
  • Validate configuration: get_key() raises a helpful error when the key is missing.
  • Expose to agents: the key is injected into os.environ for downstream usage.
tut02_3agents.py Python 
# tut02_3agents.py — environment setup
from agents import Agent, InputGuardrail, GuardrailFunctionOutput, Runner
from agents.exceptions import InputGuardrailTripwireTriggered
from pydantic import BaseModel
import os
from dotenv import load_dotenv
import asyncio

# Load API credentials from the .env file
load_dotenv()


def get_key():
    key = os.getenv('OPENAI_API_KEY')
    if not key:
        raise ValueError(
            "OPENAI_API_KEY not found. Please add it to your .env file."
        )
    return key


# Share the key with the agents runtime
os.environ['OPENAI_API_KEY'] = get_key()
